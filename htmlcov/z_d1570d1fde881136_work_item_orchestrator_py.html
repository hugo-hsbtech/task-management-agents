1"""Work Item Orchestrator — single Claude Agent SDK session driving one Linear 

2task through its full lifecycle (Linear read → Builder → Git → QA → fix loop → 

3done) per CONTEXT.md D-01. 

4 

5Architecture (RESEARCH.md Pattern 1, Pitfall 1, Pitfall 4): 

6- ONE SDK session per cycle. Skill content is injected into ``system_prompt``. 

7- Phase 2 agent modules are exposed as in-process MCP tools via 

8 ``create_sdk_mcp_server`` + ``@tool``. NO sub-agent dispatch (no 

9 ``AgentDefinition``, no ``agents=`` key in ``ClaudeAgentOptions``). 

10- Every ``@tool`` returns the canonical ``{"content":[{"type":"text","text":...}]}`` 

11 envelope; returning a Pydantic model directly silently fails the SDK serializer. 

12 

13Phase 5 (D-04, D-06): the WIO embeds skills 10 + 11 inline within the same SDK 

14session, adds a Step 1 enrichment turn before Builder and a Step 5 knowledge 

15storage turn after QA. The transition from a single ``query()`` call to a 

16multi-turn ``ClaudeSDKClient`` session preserves all of Phase 3's @tool 

17wiring (Builder/Git/QA dispatched as MCP tools). 

18 

19Two-layer QA cycle cap (CONTEXT.md D-05): 

20- Layer 1 lives in :class:`hsb.contracts.qa.QAOutput.validate_cycle_cap_logic`. 

21- Layer 2 (this module) — :func:`_check_qa_cycle_cap` — posts a Linear escalation 

22 comment when ``qa_cycle_count >= 3`` AND ``qa_status == "changes_required"``, 

23 preventing a runaway 4th cycle if Layer 1 is somehow bypassed. 

24 

25G1 (OAuth2 only): ``ANTHROPIC_API_KEY`` must be ``not in os.environ``. This 

26file does NOT have a module-top assert — that pattern broke pytest collection 

27when developer environments legitimately had the env var set. G1 is enforced 

28function-entry via :func:`hsb.agents._sdk_options.assert_oauth2_only` (called 

29from :func:`hsb.agents._sdk_options.make_options`); the SDK options here are 

30constructed via ``ClaudeAgentOptions(...)`` directly (Phase 3 pattern), but 

31the orchestrator is invoked through the global orchestrator which routes 

32through the chokepoint factory. Defensive pairing: the session-scoped 

33autouse fixture ``_gsd_clear_api_key`` in ``tests/conftest.py`` unsets the 

34env var at test session start. 

35"""

96def assemble_system_prompt() -> str: 

97 """Read all SKILL_FILES and concatenate with ``# SKILL: <stem>`` separators. 

98 

99 The headers are mandatory: without them the LLM cannot disambiguate which 

100 skill's constraints apply to which lifecycle step (RESEARCH.md 

101 Anti-Patterns). 

102 """ 

103 parts = [] 

104 for path in SKILL_FILES: 

105 content = Path(path).read_text() 

106 parts.append(f"# SKILL: {Path(path).stem}\n\n{content}") 

107 return "\n\n---\n\n".join(parts)

140 """Wrapper: delegates to Phase 2 ``builder_agent.run_builder_agent``.

157 acceptance_criteria=issue_data.get("acceptance_criteria", []),

160 repository_context=RepositoryContext( 

161 root_path=issue_data.get("root_path", ".") 

162 ), 

163 ) 

164 result = run_builder_agent(builder_input)

216async def run_orchestration_cycle(work_item_id: str | None = None) -> None: 

217 """Execute one full Work Item Orchestrator cycle. 

218 

219 Single SDK session: skill content in ``system_prompt``, Phase 2 agents as 

220 ``@tool`` wrappers. No sub-agent dispatch — only ``mcp_servers`` 

221 (CONTEXT.md D-01, RESEARCH.md Pitfall 1). 

222 

223 Parameters 

224 ---------- 

225 work_item_id: 

226 Optional Linear ID. When provided, the orchestrator drives that 

227 specific task. When ``None``, it queries Linear for the lowest-ID 

228 ``todo`` task with no unresolved dependencies and drives that one. 

229 """ 

230 system_prompt = assemble_system_prompt() 

231 

232 # CRITICAL: Do NOT register agents={} — that triggers sub-agent dispatch 

233 # (RESEARCH.md Pitfall 1). Register only mcp_servers. The allowed_tools 

234 # list must use mcp__agents__* not Task. 

235 sdk_server = create_sdk_mcp_server( 

236 name="agents", 

237 version="1.0.0", 

238 tools=[run_linear_tool, run_builder_tool, run_git_tool, run_qa_tool], 

239 ) 

240 

241 options = ClaudeAgentOptions( 

242 system_prompt=system_prompt, 

243 mcp_servers={ 

244 "agents": sdk_server, 

245 "linear": { 

246 "command": "npx", 

247 "args": ["-y", "mcp-remote", "https://mcp.linear.app/mcp"], 

248 }, 

249 }, 

250 allowed_tools=[ 

251 "mcp__agents__run_linear_op", 

252 "mcp__agents__run_builder", 

253 "mcp__agents__run_git", 

254 "mcp__agents__run_qa", 

255 "mcp__linear__get_issue", 

256 "mcp__linear__list_issues", 

257 "mcp__linear__update_issue", 

258 "mcp__linear__create_comment", 

259 ], 

260 permission_mode="acceptEdits", 

261 max_turns=30, 

262 ) 

263 

264 # work_item_id selection: if not provided, orchestrator queries Linear for 

265 # the next todo task (Phase 4 will replace this with the Global 

266 # Orchestrator selection). 

267 cycle_prompt = ( 

268 f"Run the work item lifecycle for work item {work_item_id}. " 

269 "Read its current Linear state first, then execute the next " 

270 "appropriate lifecycle step." 

271 if work_item_id 

272 else ( 

273 "Query mcp__linear__list_issues for tasks with status=todo and no " 

274 "unresolved dependencies. Select the first available task (lowest " 

275 "LIN-ID). Then run its work item lifecycle: read Linear state, " 

276 "then execute the next appropriate lifecycle step." 

277 ) 

278 ) 

279 

280 # Phase 5 lifecycle: a single ClaudeSDKClient session drives three turns 

281 # within the SAME context window: 

282 # 

283 # Step 1 [NEW Phase 5, INTL-01] — Intelligence enrichment via skill 10. 

284 # Populates `knowledge_context` for the 

285 # Builder. No Linear writes. 

286 # Steps 2-4 [Phase 3] — Builder → Git → QA cycle (max QA 

287 # cycle cap = 3, D-05 Layer 1 in 

288 # QAOutput, Layer 2 in this module). 

289 # Step 5 [NEW Phase 5, INTL-02] — Knowledge storage evaluation via 

290 # skill 11. Writes knowledge/<cat>/*.md 

291 # only if ingestion criteria are met. 

292 # 

293 # G3 runtime backstop (assert_no_task_dispatch) is called on every 

294 # received message in EVERY receive_response() loop body — catches an 

295 # SDK regression that bypasses allowed_tools at runtime (WORC-02). 

296 work_item_json = f'{{"id": "{work_item_id}"}}' if work_item_id else "{}" 

297 qa_result: dict[str, Any] = {} 

298 implementation_notes: dict[str, Any] = {} 

299 

300 async with ClaudeSDKClient(options=options) as client: 

301 # [NEW Phase 5] Step 1: Intelligence enrichment (skill 10) — before Builder. 

302 await client.query( 

303 build_enrichment_prompt(work_item_id or "next-todo", work_item_json) 

304 ) 

305 async for msg in client.receive_response(): 

306 assert_no_task_dispatch(msg) # G3 runtime backstop for G2 (WORC-02) 

307 if isinstance(msg, SystemMessage) and msg.subtype == "init": 

308 mcp_servers = msg.data.get("mcp_servers", []) 

309 # Only inspect servers we registered — the SDK init message 

310 # may surface globally-registered MCPs (e.g. user-level OAuth 

311 # servers) whose auth state is unrelated to this orchestrator. 

312 required = {"agents", "linear"} 

313 failed = [ 

314 s 

315 for s in mcp_servers 

316 if s.get("name") in required and s.get("status") != "connected" 

317 ] 

318 if failed: 

319 raise RuntimeError(f"MCP server failed to connect: {failed}") 

320 elif isinstance(msg, AssistantMessage): 

321 for block in msg.content: 

322 if isinstance(block, TextBlock): 

323 print(block.text, end="", flush=True) 

324 elif hasattr(block, "name"): 

325 logger.info("[TOOL] %s", block.name) 

326 

327 # [Phase 3] Steps 2-4: Builder → Git → QA cycle. 

328 await client.query(cycle_prompt) 

329 async for msg in client.receive_response(): 

330 assert_no_task_dispatch(msg) # G3 runtime backstop for G2 (WORC-02) 

331 if isinstance(msg, AssistantMessage): 

332 for block in msg.content: 

333 if isinstance(block, TextBlock): 

334 print(block.text) 

335 elif hasattr(block, "name"): 

336 logger.info("[TOOL] %s", block.name) 

337 elif isinstance(msg, ResultMessage): 

338 if msg.stop_reason == "error_max_turns": 

339 raise RuntimeError(f"WIO hit max_turns for {work_item_id}") 

340 # G8 context budget warning at 60% of 200K window. 

341 if msg.usage: 

342 input_tokens = msg.usage.get("input_tokens", 0) 

343 if input_tokens > 120_000: 

344 logger.warning( 

345 "WIO context at %d tokens for %s — consider skill index fallback", 

346 input_tokens, 

347 work_item_id, 

348 ) 

349 logger.info( 

350 "Orchestration cycle complete. Cost: $%.4f", 

351 msg.total_cost_usd, 

352 ) 

353 

354 # [NEW Phase 5] Step 5: Knowledge storage evaluation (skill 11) — after QA. 

355 # qa_result/implementation_notes are propagated through the system 

356 # prompt instructions; this prompt directs the LLM to pull them from 

357 # its prior turns. Side-effectful writes to knowledge/; no return. 

358 await client.query(build_storage_prompt(qa_result, implementation_notes)) 

359 async for msg in client.receive_response(): 

360 assert_no_task_dispatch(msg) # G3 runtime backstop for G2 (WORC-02) 

361 if isinstance(msg, AssistantMessage): 

362 for block in msg.content: 

363 if isinstance(block, TextBlock): 

364 print(block.text, end="", flush=True) 

365 elif hasattr(block, "name"): 

366 logger.info("[TOOL] %s", block.name) 

367 elif isinstance(msg, ResultMessage): 

368 if msg.stop_reason == "error_max_turns": 

369 raise RuntimeError( 

370 f"WIO storage step hit max_turns for {work_item_id}" 

371 ) 

372 if msg.usage: 

373 input_tokens = msg.usage.get("input_tokens", 0) 

374 if input_tokens > 120_000: 

375 logger.warning( 

376 "WIO context at %d tokens (storage step) for %s — " 

377 "consider skill index fallback", 

378 input_tokens, 

379 work_item_id, 

380 )

388async def _check_qa_cycle_cap( 

389 work_item_id: str, qa_cycle_count: int, qa_status: str 

390) -> None: 

391 """Layer 2 guardrail against QA runaway (WORC-03 / D-05 Layer 2). 

392 

393 Layer 1 (``QAOutput.validate_cycle_cap_logic`` in Phase 2) normally 

394 prevents reaching here — at ``qa_cycle_count >= 3`` Layer 1 forces 

395 ``qa_status="approved"`` with a tech-debt annotation. If somehow 

396 ``qa_cycle_count >= 3`` AND ``qa_status == "changes_required"`` still 

397 appear in the output, post a Linear escalation comment and DO NOT 

398 initiate a 4th cycle. 

399 """ 

400 if qa_cycle_count >= 3 and qa_status == "changes_required": 

401 logger.error( 

402 "SAFETY NET: qa_cycle_count=%d but qa_status=changes_required for %s. " 

403 "Escalating to human.", 

404 qa_cycle_count, 

405 work_item_id, 

406 ) 

407 await run_validated_linear_agent( 

408 operation="comment", 

409 payload={ 

410 "issueId": work_item_id, 

411 "body": ( 

412 "**Automated escalation — max QA cycles reached**\n\n" 

413 f"Max QA cycles reached (`qa_cycle_count={qa_cycle_count}`). " 

414 "Escalating to human. Task status: blocked.\n\n" 

415 "No further automated fix cycles will be initiated (WORC-03)." 

416 ), 

417 },
