123class ProviderSettings(BaseModel): 

124 """Provider configuration with discriminated auth + provider-specific configs. 

125 

126 Examples: 

127 # Claude with OAuth2 CLI (token sourced from CLAUDE_CODE_OAUTH_TOKEN) 

128 ProviderSettings( 

129 name=ProviderName.claude, 

130 model="claude-opus-4-5", 

131 auth=OAuth2CliAuth(), 

132 ) 

133 

134 # Gemini with API key 

135 ProviderSettings( 

136 name=ProviderName.gemini, 

137 model="gemini-2.5-pro", 

138 auth=ApiKeyAuth(key="..."), 

139 ) 

140 

141 # Gemini with ADC on GCP 

142 ProviderSettings( 

143 name=ProviderName.gemini, 

144 model="gemini-2.5-pro", 

145 auth=OAuth2ADCAuth(), 

146 gemini=GeminiConfig(project_id="my-gcp-project", location="us-east4"), 

147 ) 

148 """ 

149 

150 name: ProviderName = ProviderName.claude 

151 model: str = ClaudeModel.haiku_4_5 

152 auth: AuthConfig = Field(default_factory=OAuth2CliAuth) 

153 

154 # Provider-specific configs (only relevant when name matches) 

155 gemini: GeminiConfig | None = None 

156 claude: ClaudeConfig | None = None 

157 openai: OpenAIConfig | None = None 

158 

159 @model_validator(mode="after") 

160 def model_matches_provider(self) -> Self: 

161 valid: dict[ProviderName, type[StrEnum]] = { 

162 ProviderName.claude: ClaudeModel, 

163 ProviderName.openai: OpenAIModel, 

164 ProviderName.codex: CodexModel, 

165 ProviderName.gemini: GeminiModel, 

166 } 

167 allowed = set(valid[self.name]) 

168 if self.model not in allowed: 

169 raise ValueError( 

170 f"Model {self.model!r} is not valid for provider {self.name!r}. " 

171 f"Valid: {sorted(allowed)}" 

172 ) 

173 return self 

174 

175 @model_validator(mode="after") 

176 def validate_provider_config(self) -> Self: 

177 """Validate provider-specific configs match the selected provider.""" 

178 if self.gemini is not None and self.name != ProviderName.gemini: 

179 raise ValueError( 

180 f"gemini config only valid when name='gemini', got {self.name!r}" 

181 ) 

182 

183 if self.claude is not None and self.name != ProviderName.claude: 

184 raise ValueError( 

185 f"claude config only valid when name='claude', got {self.name!r}" 

186 ) 

187 

188 if self.openai is not None and self.name != ProviderName.openai: 

189 raise ValueError( 

190 f"openai config only valid when name='openai', got {self.name!r}" 

191 ) 

192 

193 # Codex check must precede the ADC check below — codex+ADC should 

194 # fail with the codex-specific message, not the gemini-only message. 

195 if self.name == ProviderName.codex and self.auth.kind != "oauth2_cli": 

196 raise ValueError( 

197 f"codex requires oauth2_cli auth (got {self.auth.kind!r}). " 

198 "Run: codex login --device-auth" 

199 ) 

200 

201 # ADC auth requires Gemini + project_id 

202 if self.auth.kind == "oauth2_adc": 

203 if self.name != ProviderName.gemini: 

204 raise ValueError(f"oauth2_adc only valid for gemini, got {self.name!r}") 

205 if self.gemini is None or self.gemini.project_id is None: 

206 raise ValueError("oauth2_adc requires gemini.project_id to be set") 

207 

208 return self
