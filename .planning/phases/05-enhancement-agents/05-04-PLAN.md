async def _detect_uat_ready_user_stories(self, linear_state: dict) -> list[dict]:
    """Detect User Stories where all child tasks are QA-approved and UAT is pending.
    Mirrors the GORD-04 EPIC-readiness check pattern. Stateless — re-runs each cycle."""
    uat_ready = []
    user_stories = [
        item for item in linear_state.values()
        if item.get("type") == "user_story"
    ]
    for us in user_stories:
        if us.get("uat_status") == "approved":
            continue
        uat_cycle_count = us.get("uat_cycle_count", 0)
        if uat_cycle_count >= 3:
            logger.warning(
                "User Story %s reached max UAT cycles (3) — escalating to human",
                us["id"],
            )
            # G6 escalation comment via Linear Agent (Plan 05-04 wires this)
            continue
        children = await self._fetch_children(us["id"])
        if children and all(c.get("qa_status") == "approved" for c in children):
            uat_ready.append(us)
    return uat_ready
```

async def get_ready_tasks(self) -> GlobalOrchestratorOutput:
    all_items = await self._fetch_all_items()
    if not all_items:
        return GlobalOrchestratorOutput(
            ready_tasks=[], is_backlog_empty=True, is_epic_ready=False,
            uat_dispatched=[], improvement_triggers=[],
        )

    # [EXISTING Phase 4] Step 1: Dependency filter
    raw_ready = self._filter_ready_items(all_items)

    # [NEW Phase 5] Step 2: Risk Agent priority queue — replaces raw_ready ordering
    risk_agent = RiskAgent()
    linear_state = {item["id"]: item for item in all_items}
    raw_task_ids = [t["id"] if isinstance(t, dict) else t.id for t in raw_ready]
    priority_queue = risk_agent.get_priority_queue(raw_task_ids, linear_state)
    ready_tasks = priority_queue.items   # Risk-sorted

    # [EXISTING Phase 4] Step 3: EPIC readiness (GORD-04)
    is_epic_ready = self._check_epic_complete(all_items)

    # [NEW Phase 5] Step 4: UAT readiness detection + dispatch
    uat_dispatched: list[str] = []
    uat_ready = await self._detect_uat_ready_user_stories(linear_state)
    for us in uat_ready:
        uat_cycle = us.get("uat_cycle_count", 0) + 1
        try:
            uat_result = await run_uat_and_validate(
                user_story_id=us["id"],
                acceptance_criteria=us.get("acceptance_criteria", []),
                uat_cycle=uat_cycle,
            )
        except RuntimeError as exc:
            logger.error("UAT Agent failed for %s: %s", us["id"], exc)
            continue   # G7: never silently continue without flagging
        # G10 pre-persist validation: B1 coverage + B3 banned tokens
        if not _uat_passes_g10(uat_result, us.get("acceptance_criteria", [])):
            logger.error("UAT G10 validation failed for %s — refusing to persist", us["id"])
            continue
        uat_dispatched.append(us["id"])
        if uat_result.overall_status == "changes_required":
            await run_validated_linear_agent(
                operation="create_subtasks",
                payload={
                    "parent_id": us["id"],
                    "findings": [s.finding for s in uat_result.scenarios if s.status == "fail"],
                },
            )
        else:
            await run_validated_linear_agent(
                operation="update",
                payload={"id": us["id"], "uat_status": "approved",
                         "uat_cycle_count": uat_cycle},
            )

    # [NEW Phase 5] Step 5: improvement triggers surfaced (no Linear writes — RISK-04)
    improvement_triggers: list[AutoImprovementTrigger] = []   # Detection deferred to operator request

    return GlobalOrchestratorOutput(
        ready_tasks=ready_tasks,
        is_backlog_empty=False,
        is_epic_ready=is_epic_ready,
        uat_dispatched=uat_dispatched,
        improvement_triggers=improvement_triggers,
    )

def _uat_passes_g10(result: UATResult, acceptance_criteria: list[str]) -> bool:
    """G10: validate UATResult before persistence.
    B1: scenario set covers all expected ACs.
    B3: no banned tokens in scenario evidence/finding (scope-creep)."""
    expected = {f"AC-{i+1}" for i in range(len(acceptance_criteria))}
    actual = {s.criterion_id for s in result.scenarios}
    if actual != expected:
        return False
    import re
    BANNED_RE = re.compile(
        r"\brefactor\b|\bcode quality\b|\bnaming\b|\bstyle\b|\blinter?\b|"
        r"\binefficient\b|\bshould also handle\b|\bfuture edge case\b",
        re.IGNORECASE,
    )
    for s in result.scenarios:
        for field in (s.evidence, s.finding or ""):
            if BANNED_RE.search(field):
                return False
    return True
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Global Orchestrator → UAT Agent (inline `await`) | The UAT Agent runs inside the Global Orchestrator's event loop (NOT a subprocess per 05-RESEARCH.md §Open Implementation Choices §4). Untrusted output (`UATResult`) crosses this boundary. Mitigated by: (1) Pydantic validation in UAT Agent's 3-retry wrapper; (2) `_uat_passes_g10` re-validates coverage and banned tokens at the Global Orchestrator BEFORE persistence (defense in depth). |
| Global Orchestrator → Linear Agent (UAT writes) | All UAT-driven Linear writes (fix subtasks, uat_status update, G6 escalation comment) go through `run_validated_linear_agent` (Phase 1 retry/backoff layer). UAT Agent itself never calls Linear. |
| Global Orchestrator → Risk Agent (priority queue) | Pure Python call; `RiskAgent.get_priority_queue()` is synchronous and deterministic — no untrusted data crosses. |
| Global Orchestrator → Linear (state read) | Phase 4 invariant: Linear is the source of truth; all reads happen through `_fetch_all_items` and `_fetch_children`, which use Linear MCP. No new attack surface in Phase 5. |
| `improvement_triggers` returned in `GlobalOrchestratorOutput` | Triggers are returned to the caller (CLI / future operator surface). They contain LLM-generated text. The caller MUST NOT auto-act on them — RISK-04 requires operator approval before any Linear write. The empty-list-by-default in this plan is the safe baseline; a future CLI command will populate the list and require explicit operator confirmation. |
